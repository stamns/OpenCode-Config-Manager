# ==================== UI 样式配置 ====================
class UIConfig:
    """全局 UI 配置"""

    # 字体配置
    FONT_FAMILY = "JetBrains Mono"
    FONT_SIZE_TITLE = 16
    FONT_SIZE_BODY = 14
    FONT_SIZE_SMALL = 12
    FONT_SIZE_NUMBER = 20

    # 配色方案 (Dark Mode)
    COLOR_BG_MAIN = "#121212"
    COLOR_BG_CARD = "#1E1E1E"
    COLOR_BG_ELEVATED = "#252525"

    COLOR_TEXT_PRIMARY = "#E0E0E0"
    COLOR_TEXT_SECONDARY = "#9E9E9E"
    COLOR_TEXT_DISABLED = "#616161"

    COLOR_BORDER = "#2D2D2D"
    COLOR_DIVIDER = "#2D2D2D"

    COLOR_PRIMARY = "#2196F3"  # 高亮/主色
    COLOR_SUCCESS = "#4CAF50"  # 成功
    COLOR_WARNING = "#FF9800"  # 警告
    COLOR_ERROR = "#F44336"  # 错误

    # 交互颜色
    COLOR_BTN_DEFAULT = "#2D2D2D"
    COLOR_BTN_HOVER = "#3A3A3A"
    COLOR_BTN_PRESSED = "#424242"

    # 布局
    WINDOW_WIDTH = 1200
    WINDOW_HEIGHT = 750

    @staticmethod
    def get_stylesheet() -> str:
        """获取全局 QSS 样式表"""
        return f"""
            /* 全局字体 */
            QWidget {{
                font-family: "{UIConfig.FONT_FAMILY}";
                font-size: {UIConfig.FONT_SIZE_BODY}px;
                color: {UIConfig.COLOR_TEXT_PRIMARY};
            }}

            /* 窗口背景 */
            QMainWindow, FluentWindow {{
                background-color: {UIConfig.COLOR_BG_MAIN};
            }}

            /* 卡片样式 */
            CardWidget, SimpleCardWidget, ElevatedCardWidget {{
                background-color: {UIConfig.COLOR_BG_CARD};
                border: 1px solid {UIConfig.COLOR_BORDER};
                border-radius: 6px;
            }}

            /* 按钮通用样式 */
            QPushButton {{
                background-color: {UIConfig.COLOR_BTN_DEFAULT};
                border: 1px solid {UIConfig.COLOR_BORDER};
                border-radius: 4px;
                padding: 6px 12px;
                color: {UIConfig.COLOR_TEXT_PRIMARY};
            }}
            QPushButton:hover {{
                background-color: {UIConfig.COLOR_BTN_HOVER};
                border-color: {UIConfig.COLOR_PRIMARY};
            }}
            QPushButton:pressed {{
                background-color: {UIConfig.COLOR_BTN_PRESSED};
            }}
            QPushButton:disabled {{
                background-color: {UIConfig.COLOR_BG_MAIN};
                color: {UIConfig.COLOR_TEXT_DISABLED};
                border-color: {UIConfig.COLOR_BORDER};
            }}

            /* 输入框 */
            QLineEdit, QTextEdit, QPlainTextEdit {{
                background-color: {UIConfig.COLOR_BTN_DEFAULT};
                border: 1px solid {UIConfig.COLOR_BORDER};
                border-radius: 4px;
                padding: 4px;
                color: {UIConfig.COLOR_TEXT_PRIMARY};
                selection-background-color: {UIConfig.COLOR_PRIMARY};
            }}
            QLineEdit:focus, QTextEdit:focus, QPlainTextEdit:focus {{
                border: 1px solid {UIConfig.COLOR_PRIMARY};
            }}

            /* 表格 */
            QTableWidget {{
                background-color: {UIConfig.COLOR_BG_CARD};
                border: 1px solid {UIConfig.COLOR_BORDER};
                gridline-color: {UIConfig.COLOR_BORDER};
            }}
            QHeaderView::section {{
                background-color: {UIConfig.COLOR_BG_ELEVATED};
                border: none;
                border-right: 1px solid {UIConfig.COLOR_BORDER};
                border-bottom: 1px solid {UIConfig.COLOR_BORDER};
                padding: 4px;
                font-weight: bold;
                color: {UIConfig.COLOR_TEXT_SECONDARY};
            }}
            QTableWidget::item {{
                padding: 4px;
            }}
            QTableWidget::item:selected {{
                background-color: {UIConfig.COLOR_BTN_HOVER};
            }}

            /* 滚动条 */
            QScrollBar:vertical {{
                background: {UIConfig.COLOR_BG_MAIN};
                width: 10px;
                margin: 0px;
            }}
            QScrollBar::handle:vertical {{
                background: {UIConfig.COLOR_BTN_DEFAULT};
                min-height: 20px;
                border-radius: 5px;
            }}
            QScrollBar::add-line:vertical, QScrollBar::sub-line:vertical {{
                height: 0px;
            }}

            /* 标签页 Pivot */
            PivotItem {{
                padding: 8px 12px;
                background: transparent;
                border-bottom: 2px solid transparent;
                color: {UIConfig.COLOR_TEXT_SECONDARY};
            }}
            PivotItem:hover {{
                color: {UIConfig.COLOR_TEXT_PRIMARY};
            }}
            PivotItem[isSelected=true] {{
                color: {UIConfig.COLOR_PRIMARY};
                border-bottom: 2px solid {UIConfig.COLOR_PRIMARY};
            }}
        """


APP_VERSION = "1.1.0"
